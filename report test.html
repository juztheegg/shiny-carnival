<script>

    // ---------- 設定區：請換成你的 Google Sheets 發佈 CSV 網址 ----------

    const REPORT_CSV_URL = 'https://docs.google.com/spreadsheets/d/e/2PACX-1vQZFPOi9JsLAfvk-rVUjcuZhltMnLphARmc39uiFAAjniiMyN9K3Q0ZwOri64AoSBJ9NOHrwUNbRQKE/pub?gid=2145666162&single=true&output=csv';   // 報告工作表

    const NAV_CSV_URL    = 'https://docs.google.com/spreadsheets/d/e/2PACX-1vQZFPOi9JsLAfvk-rVUjcuZhltMnLphARmc39uiFAAjniiMyN9K3Q0ZwOri64AoSBJ9NOHrwUNbRQKE/pub?gid=1794486467&single=true&output=csv';   // 基金淨值工作表

    // -----------------------------------------------------------------


    let reportData = [];          // 報告資料陣列 (物件陣列)

    let navData = {};             // 基金淨值資料: { 基金名稱: { dates: [], navs: [] } }

    let chartInstance = null;


    // 頁面載入時同時取得兩個 CSV

    Promise.all([

        fetch(REPORT_CSV_URL).then(res => res.text()),

        fetch(NAV_CSV_URL).then(res => res.text())

    ]).then(([reportCsv, navCsv]) => {

        parseReport(reportCsv);

        parseNav(navCsv);

        renderReportTable();

    }).catch(err => {

        console.error('讀取 CSV 失敗', err);

        alert('無法載入資料，請確認 CSV 網址是否正確，或稍後再試。');

    });


    // ---------- 解析報告 CSV ----------

    function parseReport(csv) {

        const parsed = PapaParse.parse(csv, { header: false, skipEmptyLines: true });

        const rows = parsed.data;

        // 尋找標題列：通常包含「對沖基⾦」、「投资日期」等關鍵字

        let headerRowIndex = -1;

        for (let i = 0; i < rows.length; i++) {

            const row = rows[i];

            if (row.length > 5 && row[1] && row[1].includes('對沖基⾦')) {

                headerRowIndex = i;

                break;

            }

        }

        if (headerRowIndex === -1) {

            console.warn('無法找到報告標題列，使用第一列作為標題');

            headerRowIndex = 0;

        }


        const headers = rows[headerRowIndex];

        // 資料列從 headerRowIndex + 1 開始，且第一欄必須是數字 (基金編號)

        const dataRows = [];

        for (let i = headerRowIndex + 1; i < rows.length; i++) {

            const row = rows[i];

            if (row.length === 0) continue;

            const firstCell = row[0] ? row[0].toString().trim() : '';

            if (/^\d+$/.test(firstCell)) {   // 第一欄為數字才視為基金資料列

                dataRows.push(row);

            }

        }


        // 建立物件陣列

        reportData = dataRows.map(row => {

            const obj = {};

            headers.forEach((h, idx) => {

                if (h && h.trim()) {

                    obj[h.trim()] = row[idx] ? row[idx].trim() : '';

                }

            });

            return obj;

        });

    }


    // ---------- 解析基金淨值 CSV ----------

    function parseNav(csv) {

        const parsed = PapaParse.parse(csv, { header: false, skipEmptyLines: true });

        const rows = parsed.data;

        let i = 0;

        while (i < rows.length) {

            const row = rows[i];

            if (row.length === 0) { i++; continue; }

            // 尋找基金名稱列 (例如以數字開頭且包含「基金」)

            const firstCell = row[0] ? row[0].toString().trim() : '';

            if (/^\d/.test(firstCell) && firstCell.includes('基金')) {

                // 基金名稱：去掉前面的數字與空格

                let fundName = firstCell.replace(/^\d+\s*/, '').trim();

                // 日期列：從 H 欄 (索引 7) 開始

                const dateRow = rows[i];           // 同一行，H欄以後是日期

                const navRow = rows[i + 1];         // 下一行是對應淨值

                if (!navRow) { i++; continue; }


                const dates = [];

                const navs = [];

                // 從索引 7 開始，直到遇到空值

                for (let col = 7; col < dateRow.length; col++) {

                    const dateStr = dateRow[col] ? dateRow[col].trim() : '';

                    const navStr = navRow[col] ? navRow[col].trim() : '';

                    if (dateStr && navStr && !isNaN(parseFloat(navStr))) {

                        // 嘗試解析日期 (簡單過濾)

                        if (dateStr.includes('-') || dateStr.includes('/')) {

                            dates.push(dateStr);

                            navs.push(parseFloat(navStr));

                        }

                    } else {

                        break;  // 遇到空欄位結束

                    }

                }

                if (dates.length > 0) {

                    navData[fundName] = { dates, navs };

                }

                i += 3;  // 每個基金佔用三行 (名稱行、淨值行、額外行)，跳過

            } else {

                i++;

            }

        }

    }


    // ---------- 渲染報告表格 ----------

    function renderReportTable() {

        if (reportData.length === 0) {

            document.getElementById('tableBody').innerHTML = '<tr><td colspan="10">無資料</td></tr>';

            return;

        }


        // 建立表頭 (取第一個物件的鍵，並過濾掉空欄位)

        const headers = Object.keys(reportData[0]).filter(h => h && h.trim() !== '');

        const thead = document.getElementById('tableHeader');

        thead.innerHTML = '<tr>' + headers.map(h => `<th>${h}</th>`).join('') + '</tr>';


        // 建立表身

        const tbody = document.getElementById('tableBody');

        tbody.innerHTML = reportData.map(row => {

            return '<tr>' + headers.map(h => {

                let val = row[h] || '';

                // 簡單標記負值（可依需求調整）

                if (h.includes('回報') || h.includes('盈虧') || h.includes('變動')) {

                    if (val.startsWith('-')) val = `<span class="negative">${val}</span>`;

                    else if (val && val !== '0' && val !== '0%') val = `<span class="positive">${val}</span>`;

                }

                // 將基金名稱加上可點擊連結

                if (h === '對沖基⾦') {

                    val = `<span class="fund-link" onclick="showFundDetail('${row[h]}')">${val}</span>`;

                }

                return `<td>${val}</td>`;

            }).join('') + '</tr>';

        }).join('');

    }


    // ---------- 顯示基金詳情 ----------

    window.showFundDetail = function(fundName) {

        // 從 navData 尋找匹配的基金名稱 (可能因空格或括號略有不同，用 includes 寬鬆比對)

        let matchedKey = null;

        for (let key in navData) {

            if (fundName.includes(key) || key.includes(fundName)) {

                matchedKey = key;

                break;

            }

        }

        if (!matchedKey) {

            alert('找不到該基金的淨值資料');

            return;

        }


        const fund = navData[matchedKey];

        document.getElementById('homeView').style.display = 'none';

        document.getElementById('detailView').style.display = 'block';

        document.getElementById('fundNameTitle').innerText = fundName;


        // 渲染近期淨值表格 (最近 10 筆)

        const recentTbody = document.querySelector('#recentNavTable tbody');

        const rows = [];

        for (let i = fund.dates.length - 1; i >= Math.max(0, fund.dates.length - 10); i--) {

            rows.push(`<tr><td>${fund.dates[i]}</td><td>${fund.navs[i].toFixed(2)}</td></tr>`);

        }

        recentTbody.innerHTML = rows.join('');


        // 繪製圖表

        const ctx = document.getElementById('navChart').getContext('2d');

        if (chartInstance) chartInstance.destroy();

        chartInstance = new Chart(ctx, {

            type: 'line',

            data: {

                labels: fund.dates,

                datasets: [{

                    label: '淨值',

                    data: fund.navs,

                    borderColor: 'rgb(75, 192, 192)',

                    tension: 0.1,

                    fill: false

                }]

            },

            options: {

                responsive: true,

                maintainAspectRatio: false,

                plugins: {

                    legend: { display: false },

                    tooltip: { mode: 'index', intersect: false }

                },

                scales: {

                    x: { display: true, title: { display: true, text: '日期' } },

                    y: { display: true, title: { display: true, text: '淨值' } }

                }

            }

        });

    };


    // 返回首頁

    window.showHome = function() {

        document.getElementById('homeView').style.display = 'block';

        document.getElementById('detailView').style.display = 'none';

        if (chartInstance) {

            chartInstance.destroy();

            chartInstance = null;

        }

    };

</script>
